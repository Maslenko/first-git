Примеры команд

git add readme.md        # добавляем в индекс
git commit -m "Добавил readme.md"   # коммитим
git push origin main     # отправляем на GitHub

git add .
git commit -m "Сохраняю локальные изменения перед pull"
git pull origin main

git add readme.txt       #добавить конкретный файл. 
git add .                #добавить сразу все измененные и новые файлы.
git add *.html           #добавить определенный тип файлов (например все *.html)   

nano filename — самый простой редактор (навигация стрелками, сохранить = Ctrl+O, выйти = Ctrl+X)
cat filename — просто вывести содержимое на экран
touch index.html - создать файл. В данном случае index.html
